> 调用代码实例

```
package main

import (
	"fmt"
	"github.com/loticket/verifygo"
)

func main() {
	lottery, err := verifygo.NewLottery("dlt")
	if err != nil {
		fmt.Println("不存在")
		return
	}

	lottery.CreateTicket(1, 3, "01,08,15,23,29,30-02|01,08,09", 36, 18, 1)

	_, err = lottery.Verification()
	if err != nil {
		fmt.Println(err)
		return
	}

}

```
